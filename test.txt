int cntbits(int data,char search)
{
        static int x2[32],i;
        int sum = 0;
        switch(search)
        {
        case 0:
        for (i = 0; data > 0; i++)
        {
                x2[i] = data & 1;
                data = data >> 1;
                sum = sum + !(x2[i]&1);
        }
        case 1:
        for (i = 0; data > 0; i++)
        {
                x2[i] = data & 1;
                data = data >> 1;
                sum = sum + (x2[i]&1);
        }
        }

        return sum;
}
